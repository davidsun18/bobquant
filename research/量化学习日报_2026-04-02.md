```python
# 止损止盈规则
if current_price <= entry_price * (1 - stop_loss_pct):
    trigger_stop_loss()
elif current_price >= entry_price * (1 + take_profit_pct):
    trigger_take_profit()

# 回撤止盈
if peak_price > 0:
    drawdown = (peak_price - current_price) / peak_price
    if drawdown >= max_drawdown_pct:
        trigger_trailing_stop()
```
