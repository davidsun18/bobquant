## 📚 项目 1: mlfinlab - 机器学习特征工程

### 核心学习：三重障碍法（Triple Barrier Method）

**原理**：
传统 ML 标签使用固定时间窗口（如"明天涨/跌"），但忽略了：
- 价格可能先触及止损再反弹
- 不同波动率下应使用不同的止盈/止损距离
- 时间衰减因素

三重障碍法定义三个退出边界：
1. **上障碍**: 入场价 × (1 + 动态止盈率)
2. **下障碍**: 入场价 × (1 - 动态止损率)
3. **右障碍**: 固定时间窗口（如 10 天）

标签根据价格先触及哪个障碍决定：
- 先触上障碍 → 标签 +1（做多盈利）
- 先触下障碍 → 标签 -1（做多亏损）
- 先触右障碍 → 标签 0（持有到期，按最终价格判断）

### 代码实现（适配我们的系统）

```python
# triple_barrier_labeling.py
import numpy as np
import pandas as pd
from typing import Tuple

def apply_triple_barrier(
    close: pd.Series,
    high: pd.Series,
    low: pd.Series,
    events: pd.DataFrame,
    pt_sl: Tuple[float, float] = (1.0, 1.0),
    t1: pd.Timedelta = pd.Timedelta('10D'),
    min_ret: float = 0.005
) -> pd.Series:
    """
    三重障碍法标签生成
    
    参数:
        close: 收盘价序列
        high: 最高价序列
        low: 最低价序列
        events: 事件 DataFrame，索引为入场时间，包含'close'列（入场价）
        pt_sl: (止盈倍数，止损倍数)，相对于 min_ret
        t1: 时间窗口
        min_ret: 最小预期收益率
    
    返回:
        labels: 标签序列 (+1, -1, 0)
    """
    labels = pd.Series(index=events.index, dtype='int8')
    
    for t0 in events.index:
        # 计算动态障碍
        pt = pt_sl[0] * min_ret  # 止盈阈值
        sl = pt_sl[1] * min_ret  # 止损阈值
        
        upper_barrier = events.loc[t0, 'close'] * (1 + pt)
        lower_barrier = events.loc[t0, 'close'] * (1 - sl)
        
        # 获取时间窗口内的价格路径
        t1_time = t0 + t1
        if t1_time > close.index[-1]:
            t1_time = close.index[-1]
        
        path = close.loc[t0:t1_time]
        path_high = high.loc[t0:t1_time]
        path_low = low.loc[t0:t1_time]
        
        if len(path) == 0:
            labels.loc[t0] = 0
            continue
        
        # 检查是否触及障碍
        hit_upper = (path_high >= upper_barrier).idxmax() if (path_high >= upper_barrier).any() else None
        hit_lower = (path_low <= lower_barrier).idxmax() if (path_low <= lower_barrier).any() else None
        
        # 确定退出时间
        if hit_upper is None and hit_lower is None:
            # 未触及任何障碍，持有到期
            exit_time = path.index[-1]
            exit_price = path.iloc[-1]
            ret = (exit_price - events.loc[t0, 'close']) / events.loc[t0, 'close']
            labels.loc[t0] = 1 if ret > 0 else -1 if ret < 0 else 0
        elif hit_upper is not None and (hit_lower is None or hit_upper <= hit_lower):
            # 先触及上障碍
            labels.loc[t0] = 1
        else:
            # 先触及下障碍
            labels.loc[t0] = -1
    
    return labels


def fractional_differentiation(
    price: pd.Series,
    d: float = 0.4,
    threshold: float = 1e-5
) -> pd.Series:
    """
    分数阶差分 - 在平稳性和记忆性之间取得平衡
    
    参数:
        price: 价格序列
        d: 差分阶数 (0<d<1)，越小保留越多记忆
        threshold: 权重截断阈值
    
    返回:
        分数阶差分后的序列
    """
    # 计算分数阶差分的权重
    weights = [1.0]
    for k in range(1, len(price)):
        weight = -weights[-1] * (d - k + 1) / k
        if abs(weight) < threshold:
            break
        weights.append(weight)
    
    # 应用权重
    result = pd.Series(index=price.index, dtype='float64')
    for i in range(len(weights), len(price)):
        result.iloc[i] = np.dot(weights, price.iloc[i-len(weights):i+1][::-1])
    
    return result


# 使用示例
if __name__ == '__main__':
    # 假设我们有 OHLC 数据
    df = pd.read_csv('data/000001.SZ.csv', index_col='date', parse_dates=True)
    
    # 生成事件（例如：每日开盘）

    # 应用三重障碍法
    labels = apply_triple_barrier(
        close=df['close'],
        high=df['high'],
        low=df['low'],
        events=events,
        pt_sl=(1.0, 1.0),
        t1=pd.Timedelta('5D'),  # A 股适合较短窗口
        min_ret=0.02  # 2% 最小预期收益
    )
    
    # 分数阶差分特征
    df['close_fracdiff'] = fractional_differentiation(df['close'], d=0.4)
```

```python
# feature_processors.py
from abc import ABC, abstractmethod
from typing import List, Dict, Any
import pandas as pd
import numpy as np

class Processor(ABC):
    """特征处理器基类"""
    
    @abstractmethod
    def fit(self, df: pd.DataFrame):
        pass
    
    @abstractmethod
    def transform(self, df: pd.DataFrame) -> pd.DataFrame:
        pass
    
    def fit_transform(self, df: pd.DataFrame) -> pd.DataFrame:
        self.fit(df)
        return self.transform(df)


class RobustZScoreNorm(Processor):
    """稳健 Z-Score 标准化（使用中位数和 MAD）"""
    
    def __init__(self, fields: List[str] = None):
        self.fields = fields
        self.medians = None
        self.mads = None
    
    def fit(self, df: pd.DataFrame):
        fields = self.fields or df.columns
        self.medians = df[fields].median()
        self.mads = (df[fields] - self.medians).abs().median()
    
    def transform(self, df: pd.DataFrame) -> pd.DataFrame:
        df = df.copy()
        fields = self.fields or df.columns
        # 避免除零
        mads = self.mads.replace(0, 1e-9)
        df[fields] = (df[fields] - self.medians) / mads
        return df


class CSRankNorm(Processor):
    """横截面排名归一化"""
    
    def __init__(self, fields: List[str] = None):
        self.fields = fields
    
    def fit(self, df: pd.DataFrame):
        pass  # 无需拟合
    
    def transform(self, df: pd.DataFrame) -> pd.DataFrame:
        df = df.copy()
        fields = self.fields or df.columns
        # 按日期分组，对每个截面进行排名归一化
        if 'datetime' in df.index.names:
            df[fields] = df.groupby(level='datetime')[fields].rank(pct=True)
        return df


class DropnaLabel(Processor):
    """删除标签为空的样本"""
    
    def fit(self, df: pd.DataFrame):
        pass
    
    def transform(self, df: pd.DataFrame) -> pd.DataFrame:
        if 'label' in df.columns:
            return df.dropna(subset=['label'])
        return df


# 处理器链
class ProcessorChain:
    """处理器链 - 按顺序应用多个处理器"""
    
    def __init__(self, processors: List[Processor]):
        self.processors = processors
    
    def fit(self, df: pd.DataFrame):
        for p in self.processors:
            p.fit(df)
    
    def transform(self, df: pd.DataFrame) -> pd.DataFrame:
        for p in self.processors:
            df = p.transform(df)
        return df
    
    def fit_transform(self, df: pd.DataFrame) -> pd.DataFrame:
        self.fit(df)
        return self.transform(df)


# 使用示例
if __name__ == '__main__':
    # 构建处理器链
    chain = ProcessorChain([
        RobustZScoreNorm(fields=['open', 'high', 'low', 'close', 'volume']),
        CSRankNorm(fields=['label']),
        DropnaLabel()
    ])
    
    # 应用处理
    df = pd.read_csv('data/features.csv', index_col=['datetime', 'instrument'])
    df_processed = chain.fit_transform(df)
    
    print(f"处理后数据形状：{df_processed.shape}")
```

```python
# twap_executor.py
import numpy as np
import pandas as pd
from datetime import datetime, timedelta
from typing import List, Dict
from dataclasses import dataclass

@dataclass
class TWAPOrder:
    """TWAP 订单配置"""
    symbol: str
    total_quantity: int
    duration_minutes: int
    num_slices: int  # 拆分份数
    side: str  # 'buy' or 'sell'
    
class TWAPExecutor:
    """TWAP 执行器"""
    
    def __init__(self, broker):
        self.broker = broker
        self.active_orders = {}
    
    def submit_twap(self, order: TWAPOrder) -> str:
        """
        提交 TWAP 订单
        
        返回: order_id
        """
        order_id = f"TWAP_{order.symbol}_{datetime.now().strftime('%H%M%S')}"
        
        # 计算每份数量和时间间隔
        slice_qty = order.total_quantity // order.num_slices
        remainder = order.total_quantity % order.num_slices
        interval = timedelta(minutes=order.duration_minutes / order.num_slices)
        
        # 生成执行计划
        schedule = []
        current_time = datetime.now()
        remaining_qty = order.total_quantity
        
        for i in range(order.num_slices):
            # 最后一份包含余数
            qty = slice_qty + (remainder if i == order.num_slices - 1 else 0)
            schedule.append({
                'time': current_time + i * interval,
                'quantity': qty,
                'executed': False
            })
        
        self.active_orders[order_id] = {
            'order': order,
            'schedule': schedule,
            'filled_qty': 0,
            'avg_price': 0.0,
            'status': 'active'
        }
        
        # 立即执行第一份
        self._execute_slice(order_id, 0)
        
        return order_id
    
    def _execute_slice(self, order_id: str, slice_idx: int):
        """执行单份订单"""
        if order_id not in self.active_orders:
            return
        
        order_data = self.active_orders[order_id]
        schedule = order_data['schedule']
        
        if slice_idx >= len(schedule):
            order_data['status'] = 'completed'
            return
        
        slice_info = schedule[slice_idx]
        order = order_data['order']
        
        # 提交子订单（使用限价单，价格为当前市价 ± 小偏移）
        current_price = self.broker.get_price(order.symbol)
        limit_price = current_price * (1.001 if order.side == 'buy' else 0.999)
        
        # 这里调用实际的交易接口
        # self.broker.submit_limit_order(order.symbol, order.side, slice_info['quantity'], limit_price)
        
        print(f"[TWAP] {order.symbol} {order.side} {slice_info['quantity']} @ {limit_price:.2f}")
        
        # 模拟执行（实盘时替换为实际回调）
        slice_info['executed'] = True
        order_data['filled_qty'] += slice_info['quantity']
    
    def check_and_execute(self):
        """检查并执行到期的切片"""
        current_time = datetime.now()
        
        for order_id, order_data in self.active_orders.items():
            if order_data['status'] != 'active':
                continue
            
            for i, slice_info in enumerate(order_data['schedule']):
                if not slice_info['executed'] and current_time >= slice_info['time']:
                    self._execute_slice(order_id, i)
```

#### VWAP（成交量加权平均价格）实现

```python
# vwap_executor.py
import pandas as pd
from typing import Optional

class VWAPExecutor:
    """VWAP 执行器 - 根据历史成交量分布执行"""
    
    def __init__(self, broker, lookback_days: int = 20):
        self.broker = broker
        self.lookback_days = lookback_days
        self.active_orders = {}
    
    def calculate_volume_profile(self, symbol: str) -> pd.Series:
        """
        计算历史成交量分布
        
        返回：每个时间段的成交量占比（0-1）
        """
        # 获取历史分钟线数据
        df = self.broker.get_minute_data(symbol, days=self.lookback_days)
        
        # 按时间段聚合成交量（A 股：9:30-11:30, 13:00-15:00）
        df['time_slot'] = df['datetime'].dt.time
        volume_by_slot = df.groupby('time_slot')['volume'].mean()
        
        # 归一化为占比
        volume_profile = volume_by_slot / volume_by_slot.sum()
        
        return volume_profile
    
    def submit_vwap(self, symbol: str, total_qty: int, side: str) -> str:
        """
        提交 VWAP 订单
        
        根据历史成交量分布，在成交活跃时段多执行，清淡时段少执行
        """
        order_id = f"VWAP_{symbol}_{datetime.now().strftime('%H%M%S')}"
        
        # 获取成交量分布
        volume_profile = self.calculate_volume_profile(symbol)
        
        # 根据分布分配订单量
        schedule = []
        remaining_qty = total_qty
        
        for time_slot, vol_ratio in volume_profile.items():
            # 计算该时段应执行的数量
            qty = int(total_qty * vol_ratio)
            if remaining_qty < qty:
                qty = remaining_qty
            
            schedule.append({
                'time_slot': time_slot,
                'quantity': qty,
                'executed': False
            })
            remaining_qty -= qty
        
        # 处理余数
        if remaining_qty > 0 and schedule:
            schedule[-1]['quantity'] += remaining_qty
        
        self.active_orders[order_id] = {
            'symbol': symbol,
            'side': side,
            'schedule': schedule,
            'filled_qty': 0,
            'status': 'active'
        }
        
        return order_id
```

### 风控模块设计

```python
# risk_manager.py
from dataclasses import dataclass
from typing import Dict, Optional
import numpy as np

@dataclass
class RiskLimits:
    """风险限制配置"""
    max_position_value: float  # 单只股票最大持仓市值
    max_portfolio_exposure: float  # 最大总敞口
    max_drawdown: float  # 最大回撤
    max_daily_loss: float  # 单日最大亏损
    max_order_value: float  # 单笔订单最大金额
    concentration_limit: float  # 单只股票持仓占比上限

class RiskManager:
    """风险管理器"""
    
    def __init__(self, limits: RiskLimits):
        self.limits = limits
        self.positions = {}  # symbol -> quantity
        self.prices = {}  # symbol -> current_price
        self.daily_pnl = 0.0
        self.peak_value = 0.0
        self.current_value = 0.0
    
    def update_prices(self, prices: Dict[str, float]):
        """更新价格"""
        self.prices.update(prices)
        self._recalculate_portfolio()
    
    def _recalculate_portfolio(self):
        """重新计算组合价值"""
        self.current_value = sum(
            qty * self.prices.get(symbol, 0)
            for symbol, qty in self.positions.items()
        )
        
        if self.current_value > self.peak_value:
            self.peak_value = self.current_value
    
    def check_order(self, symbol: str, side: str, quantity: int, price: float) -> tuple[bool, str]:
        """
        检查订单是否符合风控要求
        
        返回：(是否允许，原因)
        """
        order_value = quantity * price
        
        # 1. 单笔订单金额限制
        if order_value > self.limits.max_order_value:
            return False, f"订单金额 {order_value:.2f} 超过限制 {self.limits.max_order_value:.2f}"
        
        # 2. 计算执行后的持仓
        current_qty = self.positions.get(symbol, 0)
        if side == 'buy':
            new_qty = current_qty + quantity
        else:
            new_qty = current_qty - quantity
        
        new_position_value = new_qty * price
        
        # 3. 单只股票持仓限制
        if new_position_value > self.limits.max_position_value:
            return False, f"持仓市值 {new_position_value:.2f} 超过限制 {self.limits.max_position_value:.2f}"
        
        # 4. 组合总敞口限制
        current_exposure = sum(
            qty * self.prices.get(s, price) for s, qty in self.positions.items()
        )
        if side == 'buy':
            new_exposure = current_exposure + order_value
        else:
            new_exposure = current_exposure - order_value
        
        if new_exposure > self.limits.max_portfolio_exposure:
            return False, f"组合敞口 {new_exposure:.2f} 超过限制 {self.limits.max_portfolio_exposure:.2f}"
        
        # 5. 集中度限制
        if new_exposure > 0:
            concentration = new_position_value / new_exposure
            if concentration > self.limits.concentration_limit:
                return False, f"持仓集中度 {concentration:.2%} 超过限制 {self.limits.concentration_limit:.2%}"
        
        # 6. 回撤检查
        if self.peak_value > 0:
            drawdown = (self.peak_value - self.current_value) / self.peak_value
            if drawdown > self.limits.max_drawdown:
                return False, f"当前回撤 {drawdown:.2%} 超过限制 {self.limits.max_drawdown:.2%}"
        
        # 7. 单日亏损检查
        if self.daily_pnl < -self.limits.max_daily_loss:
            return False, f"单日亏损 {abs(self.daily_pnl):.2f} 超过限制 {self.limits.max_daily_loss:.2f}"
        
        return True, "通过风控检查"
    
    def update_position(self, symbol: str, quantity: int):
        """更新持仓"""
        if quantity == 0:
            self.positions.pop(symbol, None)
        else:
            self.positions[symbol] = quantity


# 使用示例
if __name__ == '__main__':
    limits = RiskLimits(
        max_position_value=500000,  # 单只股票最多 50 万

        max_order_value=100000,  # 单笔订单最多 10 万
        concentration_limit=0.30  # 单只股票最多 30%
    )
    
    risk_mgr = RiskManager(limits)
    
    # 检查订单
    allowed, reason = risk_mgr.check_order('000001.SZ', 'buy', 1000, 10.5)
    print(f"订单检查：{'✓' if allowed else '✗'} {reason}")
```
